Bootstrap toggle buttons, from Mattia Larentis

Normal Use

Html:

<div id="normal-toggle-button">
    <input id="checkbox1" type="checkbox" value="value1" checked="checked">
<div>

Js:

$('#normal-toggle-button').toggleButtons();

Callback

Html:

<div id="callback-toggle-button">
    <input id="checkbox2" type="checkbox" value="value1" checked="checked">
<div>

Js:

$('#callback-toggle-button').toggleButtons({
    onChange: function($el, status, e) {
        console.log($el, status, e);
    }
});

Size

Html:

<div id="small-toggle-button">
    <input id="checkbox3" type="checkbox" value="value1" checked="checked">
<div>

Js:

$('#small-toggle-button').toggleButtons({
    width: 100
});

Text

Html:

<div id="text-toggle-button">
    <input id="checkbox4" type="checkbox" value="value1">
<div>

Js:

$('#text-toggle-button').toggleButtons({
    label: {
        enabled: "Lorem Ipsum",
        disabled: "Dolor Sit"
    }
});

Style




Html:

<div id="danger-toggle-button">
    <input id="checkbox7" type="checkbox" value="value1" checked="checked">
</div>

Js:

$('#danger-toggle-button').toggleButtons({
    style: {
        enabled: "danger",
        disabled: "info"
    }
});
