Bootstrap toggle buttons, from Mattia Larentis

Html:

<div class="toggle-button-list">
    <input id="checkbox1" type="checkbox" value="value1" checked="checked">
    <label for="checkbox1"><label>
<div>
<div class="toggle-button-list">
    <input id="checkbox2" type="checkbox" value="value2">
    <label for="checkbox2"><label>
<div>

Js:

$('.toggle-button-list').toggleButtons();
// or
$('.toggle-button-list').toggleButtons({
    onClick: function($el, status, event) {
        console.log($el, status);
    },
    label: {
        enabled: 'YAY'
    },
    style: {
        enabled: 'primary', // 'info', 'warning', 'success' or nothing -> bootstrap buttons classes
        disabled: 'danger'
    }
});
